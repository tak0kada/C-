* 100までの素数を書き出す
```cpp
#include <iostream>
#include <vector>

bool isPrime(int n) {
    if (n == 1)
        return false;
     for (int i = 2; i < n; ++i) {
          if (n % i == 0) return false;
     }
     return true;
}

int main() {
     std::vector<int> primes;

     for (int i = 1; i < 100; ++i) {
          if (isPrime(i))
               primes.push_back(i);
     }

     for (std::vector<int>::iterator it = primes.begin(); it != primes.end(); ++it) {
          std::cout << *it << std::endl;
     }
     return 0;
}
```
